防抖 debounce必须手写

核心思路：事件触发后延迟执行，若在延迟期间再次触发则重新计时。适用于搜索输入、窗口 resize。

/**
 * 防抖：n 毫秒内连续触发只执行最后一次
 * immediate=true 时首次立即执行，之后等冷却期结束
 */
function debounce(fn, delay, immediate = false) {
  let timer = null;
  return function(...args) {
    const callNow = immediate && !timer;
    clearTimeout(timer);
    timer = setTimeout(() => {
      timer = null;
      if (!immediate) fn.apply(this, args);
    }, delay);
    if (callNow) fn.apply(this, args);
  };
}

// 使用：搜索框输入 300ms 后才发请求
const handleSearch = debounce((e) => {
  console.log('搜索：', e.target.value);
}, 300);
input.addEventListener('input', handleSearch);
节流 throttle必须手写

核心思路：固定时间间隔内只执行一次。适用于滚动监听、按钮防重复点击。

// 时间戳版：立即执行，停止触发后不再执行最后一次
function throttle(fn, interval) {
  let lastTime = 0;
  return function(...args) {
    const now = Date.now();
    if (now - lastTime >= interval) {
      lastTime = now;
      fn.apply(this, args);
    }
  };
}

// 定时器版：延迟执行，停止触发后最后一次一定会执行
function throttleTimer(fn, interval) {
  let timer = null;
  return function(...args) {
    if (!timer) {
      timer = setTimeout(() => {
        timer = null;
        fn.apply(this, args);
      }, interval);
    }
  };
}

// 使用：滚动事件每 200ms 最多触发一次
window.addEventListener('scroll', throttle(() => {
  console.log('scroll', window.scrollY);
}, 200));
⚡防抖 vs 节流口诀：防抖——等你停下来再做；节流——做了就要等一等再做。

// 手写深拷贝（支持循环引用，面试版）
function deepClone(obj, map = new WeakMap()) {
  if (obj === null || typeof obj !== 'object') return obj;
  if (map.has(obj)) return map.get(obj);   // 处理循环引用
  if (obj instanceof Date)   return new Date(obj);
  if (obj instanceof RegExp) return new RegExp(obj);

  const clone = Array.isArray(obj) ? [] : {};
  map.set(obj, clone);
  for (const key in obj) {
    if (Object.prototype.hasOwnProperty.call(obj, key)) {
      clone[key] = deepClone(obj[key], map);
    }
  }
  return clone;
}

事件委托必须手写

原理：利用事件冒泡，将子元素监听器注册到父元素，通过 event.target 判断实际触发元素。优点：减少监听器数量；动态子元素自动生效。

// 完整封装版（支持 CSS selector 匹配）
function delegate(parent, selector, eventType, handler) {
  parent.addEventListener(eventType, function(e) {
    let target = e.target;
    while (target && target !== parent) {
      if (target.matches(selector)) {
        handler.call(target, e);
        return;
      }
      target = target.parentElement;
    }
  });
}

// 使用
delegate(document.getElementById('list'), 'li', 'click', function(e) {
  console.log('点击了：', this.textContent);
});

// 面试精简版（直接写）
document.getElementById('list').addEventListener('click', (e) => {
  if (e.target.tagName === 'LI') {
    console.log(e.target.textContent);
  }
});

执行顺序：① 执行同步代码 → ② 清空全部微任务（Promise.then、queueMicrotask、MutationObserver）→ ③ 取一个宏任务执行（setTimeout、setInterval、I/O）→ 重复②③

原型链查找机制必须能解释

查找规则：访问属性时，先在对象自身找，找不到则沿 __proto__ 向上查，直到 null 为止。

关键关系：实例的 __proto__ === 构造函数的 prototype；所有对象终点是 Object.prototype.__proto__ === null

function Animal(name) { this.name = name; }
Animal.prototype.speak = function() { return `${this.name} speaks`; };

const dog = new Animal('Rex');

// 原型链：dog → Animal.prototype → Object.prototype → null
dog.__proto__ === Animal.prototype          // true
Animal.prototype.__proto__ === Object.prototype // true
Object.prototype.__proto__ === null          // true

// instanceof 本质：检查原型链上是否存在该 prototype
dog instanceof Animal  // true
dog instanceof Object  // true

Function.prototype.myBind = function(thisArg, ...bindArgs) {
  const fn = this;
  return function(...callArgs) {
    // new 调用时 this 指向新实例，忽略绑定的 thisArg
    if (new.target) return new fn(...bindArgs, ...callArgs);
    return fn.apply(thisArg, [...bindArgs, ...callArgs]);
  };
};
闭包：应用场景与内存泄漏必须能解释

定义：函数能访问定义时所在词法作用域的变量，即使函数在该作用域外执行。

// 场景1：私有变量（模块化封装）
function createCounter() {
  let count = 0;  // 私有，外部无法直接访问
  return {
    increment() { count++; },
    get()       { return count; }
  };
}

// 场景2：函数工厂（记住外部变量）
const multiply = (x) => (y) => x * y;
const double = multiply(2);
double(5); // 10

// ⚠️ 内存泄漏：闭包长期持有大量数据引用
function leak() {
  const bigData = new Array(1000000).fill('x');
  return () => console.log(bigData[0]);
  // bigData 永远不会被 GC，因为闭包持有它的引用
}
// 解决：不再需要时 bigData = null
== 隐式转换规则必须能解释
  • null == undefined → true；null/undefined 与其他值比较均为 false
  • 数字 vs 字符串：字符串转数字再比较
  • 布尔值：先转数字（true→1, false→0）再比较
  • 对象 vs 原始值：对象调用 valueOf()/toString() 转原始值
0 == ''            // true  ('' → 0)
0 == '0'           // true  ('0' → 0)
'' == '0'          // false (字符串直接比)
false == '0'      // true  (false→0, '0'→0)
null == 0         // false (特殊规则)
null == undefined  // true  (特殊规则)
[] == ![]          // true  (![]→false→0，[]→''→0，经典坑)

// 实际开发：永远用 === 避免隐式转换

二、异步处理

必考
Promise 三种状态与不可逆性
状态含义转换
pending等待中（初始）→ fulfilled 或 → rejected
fulfilled已成功不可再变化
rejected已失败不可再变化
// Promise 链式调用标准写法
new Promise((resolve, reject) => {
  setTimeout(() => resolve('data'), 1000);
})
.then(data => {
  console.log(data);   // 'data'
  return data + '!';  // 传递给下一个 then
})
.then(data => console.log(data))  // 'data!'
.catch(err => console.error(err)) // 捕获链中任何错误
.finally(() => console.log('总是执行'));
Promise.all / race / allSettled / any 对比必须能解释
方法成功条件失败条件典型场景
Promise.all全部 resolve任一 reject 立即失败并行请求，全部成功才继续
Promise.race最快的 resolve最快的 reject超时控制
Promise.allSettled全部完成（不管成败）不会失败批量请求，需要所有结果
Promise.any任一 resolve全部 reject多个备选，取最快成功
// Promise.all：并行请求多个接口
const [user, orders] = await Promise.all([
  fetch('/api/user').then(r => r.json()),
  fetch('/api/orders').then(r => r.json()),
]);

// Promise.race：实现请求超时控制
function fetchWithTimeout(url, ms) {
  const timeout = new Promise((_, reject) =>
    setTimeout(() => reject(new Error('请求超时')), ms)
  );
  return Promise.race([fetch(url), timeout]);
}
async/await 与 Promise 等价写法必须手写
// ─── Promise 链式写法 ───
function getData() {
  return fetch('/api/user')
    .then(res => res.json())
    .then(user => fetch(`/api/orders/${user.id}`))
    .then(res => res.json())
    .catch(err => console.error(err));
}

// ─── 等价的 async/await 写法 ───
async function getData() {
  try {
    const res1 = await fetch('/api/user');
    const user = await res1.json();
    const res2 = await fetch(`/api/orders/${user.id}`);
    return await res2.json();
  } catch (err) {
    console.error(err);
  }
}

// async 函数返回值永远是 Promise！
async function foo() { return 42; }
foo() instanceof Promise  // true
foo().then(v => console.log(v))  // 42
⚠️并行陷阱：循环中不要顺序 await（串行慢）
❌ for (const id of ids) { await fetch(id); }
✅ await Promise.all(ids.map(id => fetch(id)))
回调地狱 → Promise → async/await 演进原因
// ① 回调地狱：嵌套深、难读、错误处理分散
getUser(id, function(user) {
  getOrders(user.id, function(orders) {
    getDetail(orders[0].id, function(detail) {
      // 越嵌越深……
    }, onError);
  }, onError);
}, onError);

// ② Promise：链式调用，扁平化，统一 catch
getUser(id)
  .then(user => getOrders(user.id))
  .then(orders => getDetail(orders[0].id))
  .catch(onError);

// ③ async/await：像同步一样写异步，最易读，最易维护
async function main() {
  try {
    const user   = await getUser(id);
    const orders = await getOrders(user.id);
    const detail = await getDetail(orders[0].id);
  } catch(e) { onError(e); }
}

BFC 触发条件与作用必须能解释

BFC（块级格式化上下文）是独立渲染区域，内部元素不影响外部布局。

触发条件：

  • display: flex / grid / inline-block / table-cell
  • position: absolute / fixed
  • float 不为 none

解决的问题：① 清除浮动（子元素浮动导致父高度塌陷）② 阻止 margin 合并 ③ 阻止浮动元素重叠

/* 清除浮动方案二：触发 BFC */

四、HTTP 与 AJAX

必考
同源策略 + CORS 跨域原理必须能解释

同源定义：协议 + 域名 + 端口三者完全相同。

CORS 原理：服务端通过响应头 Access-Control-Allow-Origin 告诉浏览器是否允许该来源访问跨域资源。

简单请求预检请求（Preflight）
触发条件GET/POST/HEAD + 普通头PUT/DELETE 或自定义请求头
流程直接发请求先发 OPTIONS 询问，允许后发真实请求
关键头Access-Control-Allow-Origin+ Allow-Methods / Allow-Headers
// 服务端（Express）配置 CORS
app.use((req, res, next) => {
  res.setHeader('Access-Control-Allow-Origin', 'https://your-site.com');
  res.setHeader('Access-Control-Allow-Methods', 'GET,POST,PUT,DELETE');
  res.setHeader('Access-Control-Allow-Headers', 'Content-Type,Authorization');
  if (req.method === 'OPTIONS') return res.sendStatus(204); // 预检响应
  next();
});

状态码含义记忆点

type UserPreview = Pick<User, 'id' | 'name'>;
// { id: number; name: string }

// Omit<T, K>：排除指定属性（与 Pick 相反）
type UserWithoutId = Omit<User, 'id'>;
// { name: string; age: number; email: string }

// Required<T>：所有属性变必选
type StrictUser = Required<Partial<User>>; // 还原回 User
unknown vs any 的区别必须能解释
对比anyunknown
可赋值给任意类型✓✗（需类型断言/收窄）
可直接操作（调方法）✓（跳过类型检查）✗（需先收窄类型）
类型安全不安全安全
推荐场景迁移老项目临时用不知道类型时首选
// any：放弃类型检查（危险）
const a: any = 'hello';
a.toFixed(2); // 编译通过，运行时报错

// unknown：必须收窄类型才能操作（安全）
const u: unknown = 'hello';
// u.toUpperCase(); // ❌ 编译报错
if (typeof u === 'string') {
  u.toUpperCase(); // ✅ 收窄后可操作
}

// 类型断言 as（告诉 TS "我确定是这个类型"）
const el = document.getElementById('app') as HTMLDivElement;
el.style.color = 'red';
