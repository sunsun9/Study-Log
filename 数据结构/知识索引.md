## 一、线性结构

### 1.1 数组  ⭐⭐⭐

**核心原理**

连续内存，随机访问 O(1)：`address = base + index × size`

动态扩容：Java ArrayList 1.5 倍，Python list 约 2 倍。均摊插入尾部 O(1)。

**三大技巧代码（必须熟练）**

```python
# ① 双指针 —— 有序数组两数之和
def two_sum(nums, target):
    lo, hi = 0, len(nums) - 1
    while lo < hi:
        s = nums[lo] + nums[hi]
        if s == target:   return [lo, hi]
        elif s < target:  lo += 1
        else:             hi -= 1
    return []

# ② 前缀和 —— O(1) 区间求和
def build_prefix(nums):
    n = len(nums)
    pre = [0] * (n + 1)          # pre[i] = nums[0..i-1] 之和
    for i in range(n):
        pre[i + 1] = pre[i] + nums[i]
    return pre
# 区间 [l, r] 之和 = pre[r+1] - pre[l]

# ③ 差分数组 —— O(1) 区间加法，O(n) 还原
def range_add(diff, l, r, val):
    diff[l]     += val
    diff[r + 1] -= val           # 超界时需 r+1 < len(diff)

def restore(diff):
    res = diff[:]
    for i in range(1, len(res)):
        res[i] += res[i - 1]
    return res
```

**⚠️ 陷阱**
- 插入/删除中间位置是 O(n)，频繁操作中间位 → 改用链表
- 扩容时有双倍内存峰值
- 二维数组按行访问缓存命中率远高于按列访问

### 1.2 链表  ⭐⭐⭐

**核心算法代码**

```python
class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

# ① 翻转链表（迭代，三指针）⭐⭐⭐
def reverse(head):
    prev, curr = None, head
    while curr:
        nxt = curr.next   # 先存 next，防断链！
        curr.next = prev
        prev = curr
        curr = nxt
    return prev

# ② 快慢指针：找中点 ⭐⭐⭐
def find_mid(head):
    slow = fast = head
    while fast and fast.next:
        slow = slow.next
        fast = fast.next.next
    return slow  # 偶数长度时返回前半段末尾

# ③ 判断环 + 找入口（Floyd 算法）⭐⭐⭐
def detect_cycle(head):
    slow = fast = head
    while fast and fast.next:
        slow = slow.next
        fast = fast.next.next
        if slow == fast:          # 相遇，有环
            ptr = head
            while ptr != slow:    # 两指针同速，再次相遇即入口
                ptr = ptr.next
                slow = slow.next
            return ptr
    return None

# ④ 合并两个有序链表 ⭐⭐⭐
def merge_two(l1, l2):
    dummy = ListNode(0)
    cur = dummy
    while l1 and l2:
        if l1.val <= l2.val:
            cur.next = l1; l1 = l1.next
        else:
            cur.next = l2; l2 = l2.next
        cur = cur.next
    cur.next = l1 or l2
    return dummy.next

# ⑤ 删除倒数第 N 个节点（双指针）⭐⭐
def remove_nth(head, n):
    dummy = ListNode(0, head)
    fast = slow = dummy
    for _ in range(n + 1):        # fast 先走 n+1 步
        fast = fast.next
    while fast:
        fast = fast.next
        slow = slow.next
    slow.next = slow.next.next    # 跳过目标节点
    return dummy.next
```

**LRU 缓存（双向链表 + 哈希表）⭐⭐⭐**

```python
class LRUCache:
    """双向链表维护顺序，哈希表 O(1) 定位节点"""
    class Node:
        def __init__(self, k=0, v=0):
            self.key = k; self.val = v
            self.prev = self.next = None

    def __init__(self, capacity):
        self.cap = capacity
        self.cache = {}                 # key -> Node
        self.head = self.Node()         # 哑节点（最近使用端）
        self.tail = self.Node()         # 哑节点（最久未使用端）
        self.head.next = self.tail
        self.tail.prev = self.head

    def _remove(self, node):
        node.prev.next = node.next
        node.next.prev = node.prev

    def _add_front(self, node):         # 插入到 head 后（最近）
        node.next = self.head.next
        node.prev = self.head
        self.head.next.prev = node
        self.head.next = node

    def get(self, key):
        if key not in self.cache: return -1
        node = self.cache[key]
        self._remove(node)
        self._add_front(node)           # 移到最近位置
        return node.val

    def put(self, key, value):
        if key in self.cache:
            self._remove(self.cache[key])
        node = self.Node(key, value)
        self._add_front(node)
        self.cache[key] = node
        if len(self.cache) > self.cap:
            lru = self.tail.prev        # 淘汰最久未使用
            self._remove(lru)
            del self.cache[lru.key]
```

**⚠️ 陷阱**
- 翻转时**必须先存 `next`**，否则断链后找不到后继
- 边界：空链表、单节点、两节点，最容易出错
- LRU 必须同时维护链表和哈希表，缺一不可

### 1.3 栈  ⭐⭐⭐

```python
# 单调栈 —— 下一个更大元素 ⭐⭐⭐
def next_greater(nums):
    n = len(nums)
    res = [-1] * n
    stack = []                          # 存下标，维护单调递减
    for i in range(n):
        while stack and nums[stack[-1]] < nums[i]:
            res[stack.pop()] = nums[i]  # 找到了更大元素
        stack.append(i)
    return res

# 用两个栈实现队列 ⭐⭐
class MyQueue:
    def __init__(self):
        self.in_st, self.out_st = [], []

    def push(self, x):
        self.in_st.append(x)

    def pop(self):
        if not self.out_st:             # out_st 空时才倒
            while self.in_st:
                self.out_st.append(self.in_st.pop())
        return self.out_st.pop()

    def peek(self):
        val = self.pop(); self.out_st.append(val); return val
```

### 1.4 队列  ⭐⭐

```python
from collections import deque

# 滑动窗口最大值（单调队列）⭐⭐⭐
def max_sliding_window(nums, k):
    dq = deque()          # 存下标，队列内值单调递减
    res = []
    for i, v in enumerate(nums):
        while dq and nums[dq[-1]] < v:  # 维护单调性
            dq.pop()
        dq.append(i)
        if dq[0] == i - k:             # 队头超出窗口
            dq.popleft()
        if i >= k - 1:
            res.append(nums[dq[0]])    # 队头就是最大值
    return res
```

**⚠️ 陷阱**
- BFS **必须在入队时标记已访问**，不能在出队时标记（否则重复入队）
- 循环队列判满：`(rear + 1) % cap == front`（留一个空位区分空和满）

## 二、树 Tree

### 2.1 二叉树遍历  ⭐⭐⭐

```python
class TreeNode:
    def __init__(self, val=0, left=None, right=None):
        self.val = val; self.left = left; self.right = right

# 迭代中序遍历（面试最常考）⭐⭐⭐
def inorder(root):
    stack, res, curr = [], [], root
    while curr or stack:
        while curr:                     # 一路向左
            stack.append(curr)
            curr = curr.left
        curr = stack.pop()
        res.append(curr.val)
        curr = curr.right               # 转向右子树
    return res

# 层序遍历 BFS ⭐⭐⭐
def level_order(root):
    if not root: return []
    q, res = deque([root]), []
    while q:
        level = []
        for _ in range(len(q)):         # 每轮处理一层
            node = q.popleft()
            level.append(node.val)
            if node.left:  q.append(node.left)
            if node.right: q.append(node.right)
        res.append(level)
    return res
```

**前序/中序重建二叉树 ⭐⭐**

```python
def build_tree(preorder, inorder):
    if not preorder: return None
    root_val = preorder[0]
    idx = inorder.index(root_val)       # 中序中根的位置
    root = TreeNode(root_val)
    root.left  = build_tree(preorder[1:1+idx], inorder[:idx])
    root.right = build_tree(preorder[1+idx:],  inorder[idx+1:])
    return root
```

> 记住：前序+中序 可唯一重建；后序+中序 可唯一重建；**前序+后序 不可唯一重建**

```python
# 验证 BST（传上下界，不能只看父子）⭐⭐⭐
def is_valid_bst(root, lo=float('-inf'), hi=float('inf')):
    if not root: return True
    if not (lo < root.val < hi): return False
    return (is_valid_bst(root.left, lo, root.val) and

# BST 第 K 小元素（中序第 K 个）⭐⭐
def kth_smallest(root, k):
    stack, n = [], 0
    curr = root
    while curr or stack:
        while curr:
            stack.append(curr); curr = curr.left
        curr = stack.pop()
        n += 1
        if n == k: return curr.val
        curr = curr.right
```

```python
def lca(root, p, q):
    if not root or root == p or root == q:
        return root
    left  = lca(root.left, p, q)
    right = lca(root.right, p, q)
    if left and right: return root      # p、q 分属两侧，根就是 LCA
```

### 2.4 二叉树最大路径和  ⭐⭐⭐

```python
def max_path_sum(root):
    res = [float('-inf')]
    def dfs(node):
        if not node: return 0
        l = max(dfs(node.left), 0)      # 负数不要
        r = max(dfs(node.right), 0)
        res[0] = max(res[0], l + r + node.val)  # 过当前节点的最大路径
        return max(l, r) + node.val     # 向上只能选一侧
    dfs(root)
    return res[0]
```

### 2.5 红黑树  ⭐⭐

**五条性质（背诵）**

1. 节点是红色或黑色
2. 根节点是黑色
3. 所有叶节点（NIL）是黑色
4. 红色节点的子节点都是黑色（不能连续红）

| 对比项 | AVL 树 | 红黑树 |
|--------|--------|--------|
| 平衡程度 | 严格平衡 | 近似平衡 |
| 插入/删除旋转 | 较多 | 最多 3 次 |
| 适用场景 | 读多写少 | 读写均衡 |
| 实际应用 | — | Java TreeMap、C++ map、Linux CFS |

> 面试不要求手写旋转，但要能解释 5 条性质和"变色→旋转"修复策略

### 2.6 B+ 树  ⭐⭐

| 对比项 | B 树 | B+ 树 |
|--------|------|-------|
| 数据位置 | 所有节点 | 只在叶子 |
| 叶子连接 | 无 | 双向链表 |
| 范围查询 | 需要回溯 | 顺序遍历链表 |

**MySQL InnoDB 选 B+ 树的三个原因**：
1. 内节点只存 key，树更矮，减少磁盘 I/O

3. 全表扫描只需遍历叶子层

## 三、堆 Heap

### 原理 + 复杂度  ⭐⭐⭐

- 完全二叉树，数组存储：父 `i` → 左 `2i+1`，右 `2i+2`；子 `i` → 父 `(i-1)//2`
- 建堆：O(n)（不是 O(n log n)！）—— **高频考点**
- 插入/删除堆顶：O(log n)

### 手写最小堆  ⭐⭐⭐

```python
class MinHeap:
    def __init__(self):
        self.h = []

    def push(self, val):
        self.h.append(val)
        self._sift_up(len(self.h) - 1)

    def pop(self):
        self.h[0], self.h[-1] = self.h[-1], self.h[0]
        val = self.h.pop()
        if self.h: self._sift_down(0)
        return val

    def _sift_up(self, i):
        while i > 0:
            p = (i - 1) // 2
            if self.h[p] > self.h[i]:
                self.h[p], self.h[i] = self.h[i], self.h[p]
                i = p
            else: break

    def _sift_down(self, i):
        n = len(self.h)
        while True:
            smallest = i
            for c in [2*i+1, 2*i+2]:
                if c < n and self.h[c] < self.h[smallest]:
                    smallest = c
            if smallest == i: break
            self.h[i], self.h[smallest] = self.h[smallest], self.h[i]
            i = smallest
```

### Top-K 问题  ⭐⭐⭐

```python
import heapq

# 求最大的 K 个元素 → 维护大小为 K 的最小堆
def top_k(nums, k):
    heap = []
    for x in nums:
        heapq.heappush(heap, x)
        if len(heap) > k:
            heapq.heappop(heap)         # 弹出最小，保留最大 K 个
    return heap

# 数据流中位数（大根堆 + 小根堆）⭐⭐⭐
class MedianFinder:
    def __init__(self):
        self.lo = []   # 最大堆（存较小半部分，取负模拟）
        self.hi = []   # 最小堆（存较大半部分）

    def add_num(self, num):
        heapq.heappush(self.lo, -num)
        heapq.heappush(self.hi, -heapq.heappop(self.lo))
        if len(self.hi) > len(self.lo):
            heapq.heappush(self.lo, -heapq.heappop(self.hi))

    def find_median(self):
        if len(self.lo) > len(self.hi):
            return -self.lo[0]
        return (-self.lo[0] + self.hi[0]) / 2
```

**⚠️ 陷阱**
- Python `heapq` 是最小堆，最大堆需存负数
- Top-K 最大元素用**最小堆**而非最大堆（能快速淘汰小元素）
- 建堆 O(n) vs 逐个插入 O(n log n)，区别在于 sift-down 从底往上代价更低

## 四、哈希表

### 原理与冲突解决  ⭐⭐

- **链地址法**（Java HashMap）：链表长度 > 8 且数组长度 ≥ 64 → 转红黑树
- **开放地址法**：线性探测（聚集问题）→ 二次探测 → 双重哈希

**Java HashMap 关键参数**：初始容量 16，负载因子 0.75，扩容翻倍，重新哈希

### 滑动窗口（哈希表高频应用）  ⭐⭐⭐

```python
# 最长无重复子串 ⭐⭐⭐
def length_of_longest_substring(s):
    pos = {}              # char -> 最后出现的下标
    lo, res = 0, 0
    for hi, c in enumerate(s):
        if c in pos and pos[c] >= lo:
            lo = pos[c] + 1     # 左指针跳过重复字符
        pos[c] = hi
        res = max(res, hi - lo + 1)
    return res

# 最小覆盖子串 ⭐⭐⭐
def min_window(s, t):
    from collections import Counter
    need = Counter(t)
    missing = len(t)    # 还需要补充的字符总数
    lo = start = end = 0
    for hi, c in enumerate(s, 1):
        if need[c] > 0:
            missing -= 1
        need[c] -= 1
        if missing == 0:
            while need[s[lo]] < 0:   # 收缩左边
                need[s[lo]] += 1
                lo += 1
            if not end or hi - lo < end - start:
                start, end = lo, hi
            need[s[lo]] += 1         # 准备下次扩展
            missing += 1
            lo += 1
    return s[start:end]
```

**⚠️ 陷阱**
- 自定义对象做 key 必须同时重写 `hashCode()` 和 `equals()`
- 多线程用 `ConcurrentHashMap`，不要用 `HashTable`（全局锁）
- Java 7 并发扩容有死循环风险（链表成环），Java 8 尾插法修复

## 五、图 Graph

### DFS / BFS 模板  ⭐⭐⭐

```python
from collections import deque

# DFS 递归
def dfs(node, visited, graph):
    visited.add(node)
    for nb in graph[node]:
        if nb not in visited:
            dfs(nb, visited, graph)

# BFS —— 无权最短路
def bfs(start, graph):
    dist = {start: 0}
    q = deque([start])
    while q:
        node = q.popleft()
        for nb in graph[node]:
            if nb not in dist:
                dist[nb] = dist[node] + 1
                q.append(nb)     # 入队时标记，不是出队时！
    return dist
```

### Dijkstra 最短路  ⭐⭐⭐

```python
import heapq

def dijkstra(graph, start):
    # graph[u] = [(v, weight), ...]
    dist = {start: 0}
    pq = [(0, start)]               # (距离, 节点)
    while pq:
        d, u = heapq.heappop(pq)
        if d > dist.get(u, float('inf')): continue  # 过期条目
        for v, w in graph.get(u, []):
            nd = d + w
            if nd < dist.get(v, float('inf')):
                dist[v] = nd
                heapq.heappush(pq, (nd, v))
    return dist
```

> Dijkstra 不能处理负权边！负权边 → 用 Bellman-Ford

### 拓扑排序 Kahn 算法  ⭐⭐⭐

```python
def topo_sort(n, edges):
    graph = [[] for _ in range(n)]
    in_degree = [0] * n
    for u, v in edges:
        graph[u].append(v)
        in_degree[v] += 1

    q = deque(i for i in range(n) if in_degree[i] == 0)
    order = []
    while q:
        u = q.popleft()
        order.append(u)
        for v in graph[u]:
            in_degree[v] -= 1
            if in_degree[v] == 0:
                q.append(v)

    return order if len(order) == n else []  # 长度不足 → 有环
```

### 并查集 Union-Find  ⭐⭐⭐

```python
class UnionFind:
    def __init__(self, n):
        self.parent = list(range(n))
        self.rank = [0] * n
        self.count = n              # 连通分量数

    def find(self, x):              # 路径压缩
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])
        return self.parent[x]

    def union(self, x, y):          # 按秩合并
        px, py = self.find(x), self.find(y)
        if px == py: return False
        if self.rank[px] < self.rank[py]: px, py = py, px
        self.parent[py] = px
        if self.rank[px] == self.rank[py]: self.rank[px] += 1
        self.count -= 1
        return True

    def connected(self, x, y):
        return self.find(x) == self.find(y)
```

**⚠️ 陷阱**
- 有向图判环：三色标记（白=未访问，灰=访问中，黑=已完成），访问到灰节点 → 有环
- 无向图判环：访问到已访问节点且非父节点 → 有环
- Dijkstra + 负权边 = 错误结果（面试最常考的陷阱）
- 并查集只能合并，不支持拆分

## 六、字符串与 Trie

### KMP 算法  ⭐⭐

```python
def kmp(text, pattern):
    # 构建 next 数组（最长公共前后缀）
    def build_next(p):
        nxt = [0] * len(p)
        j = 0
        for i in range(1, len(p)):
            while j > 0 and p[i] != p[j]:
                j = nxt[j - 1]          # 回退
            if p[i] == p[j]:
                j += 1
            nxt[i] = j
        return nxt

    nxt = build_next(pattern)
    j = 0
    res = []
    for i, c in enumerate(text):
        while j > 0 and c != pattern[j]:
            j = nxt[j - 1]
        if c == pattern[j]:
            j += 1
        if j == len(pattern):
            res.append(i - j + 1)       # 匹配成功
            j = nxt[j - 1]
    return res
```

### Trie 前缀树  ⭐⭐

```python
class TrieNode:
    def __init__(self):
        self.children = {}
        self.is_end = False

class Trie:
    def __init__(self):
        self.root = TrieNode()

    def insert(self, word):
        node = self.root
        for c in word:
            if c not in node.children:
                node.children[c] = TrieNode()
            node = node.children[c]
        node.is_end = True

    def search(self, word):             # 完整单词
        node = self._find(word)
        return node is not None and node.is_end

    def starts_with(self, prefix):      # 前缀查找
        return self._find(prefix) is not None

    def _find(self, s):
        node = self.root
        for c in s:
            if c not in node.children: return None
            node = node.children[c]
        return node
```

操作时间复杂度：O(L)，L 为字符串长度，与词典大小无关。

## 七、排序算法

### 复杂度总览

| 算法 | 平均 | 最坏 | 空间 | 稳定 | 备注 |
|------|------|------|------|------|------|
| 快速排序 | O(n log n) | O(n²) | O(log n) | ❌ | 通用首选 ⭐⭐⭐ |
| 归并排序 | O(n log n) | O(n log n) | O(n) | ✅ | 稳定首选 ⭐⭐⭐ |
| 堆排序 | O(n log n) | O(n log n) | O(1) | ❌ | 空间受限 ⭐⭐ |
| 插入排序 | O(n²) | O(n²) | O(1) | ✅ | 近乎有序时接近 O(n) |
| 计数/基数/桶 | O(n+k) | — | O(n+k) | ✅ | 非比较排序，有条件 ⭐⭐ |

### 快速排序  ⭐⭐⭐

```python
import random

def quick_sort(arr, lo, hi):
    if lo >= hi: return
    p = partition(arr, lo, hi)
    quick_sort(arr, lo, p - 1)
    quick_sort(arr, p + 1, hi)

def partition(arr, lo, hi):
    # 随机化 pivot，避免有序数组退化 O(n²)
    rand = random.randint(lo, hi)
    arr[rand], arr[hi] = arr[hi], arr[rand]
    pivot = arr[hi]
    i = lo - 1
    for j in range(lo, hi):
        if arr[j] <= pivot:
            i += 1
            arr[i], arr[j] = arr[j], arr[i]
    arr[i+1], arr[hi] = arr[hi], arr[i+1]
    return i + 1

# 三路快排（大量重复元素时退化为 O(n)）⭐⭐
def three_way_partition(arr, lo, hi):
    pivot = arr[lo]
    lt, gt, i = lo, hi, lo + 1
    while i <= gt:
        if   arr[i] < pivot: arr[lt], arr[i] = arr[i], arr[lt]; lt += 1; i += 1
        elif arr[i] > pivot: arr[gt], arr[i] = arr[i], arr[gt]; gt -= 1
        else: i += 1
    return lt, gt
```

### 归并排序  ⭐⭐⭐

```python
def merge_sort(arr):
    if len(arr) <= 1: return arr
    mid = len(arr) // 2
    l = merge_sort(arr[:mid])
    r = merge_sort(arr[mid:])
    return merge(l, r)

def merge(l, r):
    res, i, j = [], 0, 0
    while i < len(l) and j < len(r):
        if l[i] <= r[j]:              # <= 保证稳定性
            res.append(l[i]); i += 1
        else:
            res.append(r[j]); j += 1
    return res + l[i:] + r[j:]
```

### 数组第 K 大元素（快速选择）  ⭐⭐⭐

```python
def find_kth_largest(nums, k):
    # 快速选择，平均 O(n)，无需完整排序
    lo, hi, target = 0, len(nums) - 1, len(nums) - k
    while lo < hi:
        p = partition(nums, lo, hi)
        if   p < target: lo = p + 1
        elif p > target: hi = p - 1
        else: break
    return nums[target]
```

**⚠️ 陷阱**
- Java `Arrays.sort` 基本类型用双轴快排（不稳定），对象用 TimSort（稳定）
- 快排 pivot 选最后一个元素，有序输入会退化 → **必须随机化**
- 计数/基数/桶排序突破 O(n log n) 下界，但有使用前提（整数范围、均匀分布等）

## 八、查找算法

### 二分查找  ⭐⭐⭐

```python
# 基础版
def binary_search(arr, target):
    lo, hi = 0, len(arr) - 1
    while lo <= hi:
        mid = lo + (hi - lo) // 2    # 防 lo+hi 溢出
        if arr[mid] == target:   return mid
        elif arr[mid] < target:  lo = mid + 1
        else:                    hi = mid - 1
    return -1

# 找第一个 >= target 的位置（左边界）⭐⭐⭐
def lower_bound(arr, target):
    lo, hi = 0, len(arr)
    while lo < hi:
        mid = (lo + hi) // 2
        if arr[mid] < target: lo = mid + 1
        else:                 hi = mid
    return lo                        # lo == hi，即答案位置

# 找最后一个 <= target 的位置（右边界）
def upper_bound(arr, target):
    lo, hi = 0, len(arr)
    while lo < hi:
        mid = (lo + hi) // 2
        if arr[mid] <= target: lo = mid + 1
        else:                  hi = mid
    return lo - 1

# 旋转数组查找 ⭐⭐⭐
def search_rotated(nums, target):
    lo, hi = 0, len(nums) - 1
    while lo <= hi:
        mid = (lo + hi) // 2
        if nums[mid] == target: return mid
        if nums[lo] <= nums[mid]:              # 左半有序
            if nums[lo] <= target < nums[mid]: hi = mid - 1
            else:                              lo = mid + 1
        else:                                  # 右半有序
            if nums[mid] < target <= nums[hi]: lo = mid + 1
            else:                              hi = mid - 1
    return -1
```

**⚠️ 陷阱**
- `mid = (lo + hi) / 2` 在 lo+hi 溢出时出错，用 `lo + (hi - lo) / 2`
- 二分不只用于有序数组，**只要具备单调性就可以用**（如求最小化最大值）

## 九、动态规划

### 核心四要素  ⭐⭐⭐

1. **状态定义**：dp[i] 表示什么
2. **转移方程**：dp[i] 如何从子状态推导
3. **初始化**：边界条件
4. **遍历顺序**：依赖哪些之前的状态

### 经典模板代码

```python
# 最长递增子序列 LIS ⭐⭐⭐
def lis(nums):
    dp = []                         # dp[i] = 长度为 i+1 的 LIS 末尾最小值
    for x in nums:
        lo, hi = 0, len(dp)
        while lo < hi:              # 二分找插入位置，O(n log n)
            mid = (lo + hi) // 2
            if dp[mid] < x: lo = mid + 1
            else:           hi = mid
        if lo == len(dp): dp.append(x)
        else:             dp[lo] = x
    return len(dp)

# 最长公共子序列 LCS ⭐⭐⭐
def lcs(s, t):
    m, n = len(s), len(t)
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if s[i-1] == t[j-1]:
                dp[i][j] = dp[i-1][j-1] + 1
            else:
                dp[i][j] = max(dp[i-1][j], dp[i][j-1])
    return dp[m][n]

# 编辑距离 ⭐⭐⭐
def edit_distance(s, t):
    m, n = len(s), len(t)
    dp = list(range(n + 1))         # 空间压缩为一维
    for i in range(1, m + 1):
        prev = dp[0]
        dp[0] = i
        for j in range(1, n + 1):
            temp = dp[j]
            if s[i-1] == t[j-1]:
                dp[j] = prev
            else:
                dp[j] = 1 + min(prev, dp[j], dp[j-1])
            prev = temp
    return dp[n]

# 0-1 背包 ⭐⭐⭐
def knapsack_01(weights, values, W):
    dp = [0] * (W + 1)
    for w, v in zip(weights, values):
        for j in range(W, w - 1, -1):   # 从大到小，防止重复选
            dp[j] = max(dp[j], dp[j - w] + v)
    return dp[W]

# 完全背包（每件物品可取多次）
def knapsack_complete(weights, values, W):
    dp = [0] * (W + 1)
    for w, v in zip(weights, values):
        for j in range(w, W + 1):       # 从小到大，允许重复选
            dp[j] = max(dp[j], dp[j - w] + v)
    return dp[W]

# 最大子数组和 Kadane 算法 ⭐⭐⭐
def max_subarray(nums):
    cur = res = nums[0]
    for x in nums[1:]:
        cur = max(x, cur + x)
        res = max(res, cur)
    return res
```

**⚠️ 陷阱**
- 0-1 背包**从大到小**遍历容量；完全背包**从小到大**
- 区分"子序列"（不连续）和"子数组/子串"（连续），状态定义完全不同
- 空间优化（二维→一维）时确认遍历方向不能颠倒
